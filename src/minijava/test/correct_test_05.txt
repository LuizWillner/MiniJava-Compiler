class TesteArray {
    public static void main(String[] args) {
        System.out.println(new ArrayOps().somaElementos());
    }
}

class ArrayOps {
    public int somaElementos() {
        int[] numeros;
        int soma;
        int i;

        numeros = new int[5];
        numeros[0] = 10;
        numeros[1] = 20;
        numeros[2] = 30;
        numeros[3] = 40;
        numeros[4] = 50;

        soma = 0;
        i = 0;

        while (i < numeros.length) {
            soma = soma + numeros[i];
            i = i + 1;
        }

        return soma;
    }
}
